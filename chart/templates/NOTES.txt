Gateway List Service has been deployed!

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.api.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "gateway-list-service.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.api.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "gateway-list-service.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.api.service.port }}
{{- else if contains "ClusterIP" .Values.api.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "gateway-list-service.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=api" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:3000 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:3000
{{- end }}

2. Check API health:
  kubectl --namespace {{ .Release.Namespace }} exec -it $(kubectl get pods -n {{ .Release.Namespace }} -l "app.kubernetes.io/component=api" -o jsonpath="{.items[0].metadata.name}") -- curl http://localhost:3000/health

3. Check worker CronJob:
  kubectl --namespace {{ .Release.Namespace }} get cronjob {{ include "gateway-list-service.fullname" . }}-worker

4. View worker logs:
  kubectl --namespace {{ .Release.Namespace }} logs -l app.kubernetes.io/component=worker --tail=100

5. Test V3 gateway endpoint:
  curl -X POST https://{{ (index .Values.ingress.hosts 0).host }}/v3/gateway \
    -H "Content-Type: application/json" \
    -d '{"site":"site1"}'
